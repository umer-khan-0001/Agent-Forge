Quick Example

Build an autonomous research agent in minutes

from agentforge import Agent
from agentforge.tools import WebSearchTool, CalculatorTool

# Initialize agent with tools
agent = Agent(
    name="ResearchAssistant",
    llm_model="gpt-4",
    tools=[WebSearchTool(), CalculatorTool()],
    memory_enabled=True
)

# Run autonomous task
response = agent.run(
    "Find the current price of Tesla stock and calculate "
    "the percentage change from its IPO price of $17"
)

print(response.final_answer)
# Output: "Tesla (TSLA) is currently trading at $248.50. 
#          From its IPO price of $17, this represents 
#          a 1361% increase."
